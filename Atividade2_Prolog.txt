interface("ItfAluno").
interface("ItfProfessor").
interface("ItfCoordenador").

classe("ClsPessoa").
classe("ClsAluno").
classe("ClsProfessor").

implementa("ClsPessoa", "ItfPessoa").

herda("ClsProfessor", "ClsPessoa").

/* 2) Regras */
subclasse(X, Y) :- classe(X), herda(X, Y), classe(Y).
contrata(X, Y) :- classe(X), implementa(X, Y), interface(Y).
